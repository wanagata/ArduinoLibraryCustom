#include "Arduino.h"
#include "$.h"

$::$(int pin)
{
  pinMode(pin, OUTPUT);
  _pin = pin;
}

void $::begin()
{
  Serial.println("$ class begin!");
}

void $::loop()
{
  Serial.println("$ class in loop function!");
  $::dot(); $::dot(); $::dot();
  $::dash(); $::dash(); $::dash();
  $::dot(); $::dot(); $::dot();
}

void $::dot()
{
  digitalWrite(_pin, HIGH);
  delay(250);
  digitalWrite(_pin, LOW);
  delay(250);  
}

void $::dash()
{
  digitalWrite(_pin, HIGH);
  delay(1000);
  digitalWrite(_pin, LOW);
  delay(250);
}
